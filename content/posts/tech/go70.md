- 进行统一封装
  1. 前后端协作效率：一致的响应格式让前端开发者能以统一的方式处理服务端响应
  2. 错误处理简化：标准化的错误码和消息便于统一处理各种异常情况
  3. 接口文档维护：规范化响应减少文档编写工作量
  4. 客户端适配：移动端或其他客户端可以复用相同的响应解析逻辑

``` go
// 一个 Gin 中间件，用于注入 TraceID 和最后打印请求
func NewTrace() gin.HandlerFunc {
    return func(c *gin.Context) {
        start := time.Now()

        // 优先从请求头获取 TraceID，没有则生成新的
        traceID := c.GetHeader("X-Trace-ID")
        if traceID == "" {
            traceID = uuid.New().String()
        }
        // tag: c.Request.Header.Get("Authorization") jwt的信息在权限验证中间件使用
        // tag: gin中间件传递参数 Gin 的 c.Set 只在 Gin 内部生效
        tmpParam := c.GetHeader("token")
        c.Set("user", tmpParam) // 可以先设置 如果有jwt中间件的就解析后再设置

        // 将 TraceID 注入到标准的 context.Context 中
        // slog 需要标准库的 Context
        ctx := context.WithValue(c.Request.Context(), constant.TraceIDKey, traceID) // "trace_id": "xxx" 绑定到该ctx了
        // 补充设置头信息
        c.Request = c.Request.WithContext(ctx)

        // tag: 将 TraceID 写入响应头，方便客户端追踪
        c.Header("X-Trace-ID", traceID)

        c.Next()

        // 请求结束后的汇总日志 日志等级info
        slog.InfoContext(c.Request.Context(), "请求信息",
            slog.String("method", c.Request.Method),
            slog.String("path", c.Request.URL.Path),
            slog.Int("status", c.Writer.Status()),
            slog.Int("body_size", c.Writer.Size()),
            slog.Duration("latency", time.Since(start)),
        )
    }
}
```

### 统一拦截异常进行处理

1. 连接中断处理：特别处理了 "broken pipe" 和 "connection reset by peer" 错误，避免客户端提前断开连接时产生冗余错误日志
2. 错误信息记录：记录错误详情和堆栈信息，便于问题排查
3. 统一错误响应：所有异常都以统一格式返回给客户端
``` go
// 自定义异常恢复中间件
func NewCustomRecovery() gin.HandlerFunc {
    return func(c *gin.Context) {
        defer func() {
            if err := recover(); err != nil {
                var brokenPipe bool
                // 检测是否是连接中断
                if ne, ok := err.(*net.OpError); ok {
                    if se, ok := ne.Err.(*os.SyscallError); ok {
                        if strings.Contains(strings.ToLower(se.Error()), "broken pipe") ||
                            strings.Contains(strings.ToLower(se.Error()), "connection reset by peer") {
                            brokenPipe = true
                        }
                    }
                }

                // 获取堆栈信息
                stack := string(debug.Stack())

                // 获取原始请求内容（可选，方便排查是哪个参数导致的崩溃）
                httpRequest, _ := httputil.DumpRequest(c.Request, false)

                if brokenPipe {
                    slog.ErrorContext(c.Request.Context(), "网络连接中断",
                        slog.Any("error", err),
                        slog.String("request", string(httpRequest)),
                    )
                    c.Abort()
                    response.InternalServerError(c, nil, &response.ResponseOption{Msg: "network abort"})
                    return
                }

                // tag: 获取traceId
                ctx := c.Request.Context()
                traceID, _ := ctx.Value(constant.TraceIDKey).(string)
                // 除了打印 看是否赋值到返回json里
                slog.Error("exception catch", "error", err, "stack", stack, "traceID", traceID, "request", string(httpRequest))

                // c.AbortWithStatusJSON()
                c.Abort()
                response.InternalServerError(c, nil, &response.ResponseOption{Msg: "server internal error"})
            }
        }()
        c.Next()
    }
}
```

``` go
    r.GET("get1", func(c *gin.Context) {

    r.GET("get2/:id/:name", func(c *gin.Context) {
        id := c.Param("id") // tag: Param 只获取路径参数（如 /post2/:id

    // tag: form表单提交
```

## 参数验证
- 常用规则
```
// 不能为空，并且不能没有这个字段
required： 必填字段，如：binding:"required"

// 针对字符串的长度
min 最小长度，如：binding:"min=5"
max 最大长度，如：binding:"max=10"
len 长度，如：binding:"len=6"

// 针对数字的大小
eq 等于，如：binding:"eq=3"
ne 不等于，如：binding:"ne=12"
gt 大于，如：binding:"gt=10"
```

``` go
var trans ut.Translator

func init() {
    // 创建翻译器 使用了中文包
    uni := ut.New(zh.New())
    trans, _ = uni.GetTranslator("zh")

    // 注册翻译器
    v, ok := binding.Validator.Engine().(*validator.Validate)
    if ok {
        _ = zh_translations.RegisterDefaultTranslations(v, trans)
    }

    // tag: 注册新标签 反射拿标签label
    v.RegisterTagNameFunc(func(field reflect.StructField) string {
        label := field.Tag.Get("label")
        if label == "" {
            label = field.Name
        }
        name := field.Tag.Get("json")
        return fmt.Sprintf("%s---%s", name, label)
    })

    // tag: 自定义函数验证规则 验证ip地址
    v.RegisterValidation("fip", func(fl validator.FieldLevel) bool {
        fmt.Println("fl.Field(): ", fl.Field())
        fmt.Println("fl.FieldName(): ", fl.FieldName())
        fmt.Println("fl.StructFieldName(): ", fl.StructFieldName())
        fmt.Println("fl.Parent(): ", fl.Parent())
        fmt.Println("fl.Top(): ", fl.Top())
        fmt.Println("fl.Param(): ", fl.Param())

        ip, ok := fl.Field().Interface().(string)
        if ok && ip != "" {
            // 传了值就去校验是不是IP地址
            ipObj := net.ParseIP(ip)
            return ipObj != nil // 是否正常返回id对象
        }
        return true
    })

    // fl.Field():  1232
    // fl.FieldName():  ip---ip地址
    // fl.StructFieldName():  Ip
    // fl.Parent():  { 12121@163.com 1232}
    // fl.Top():  &{ 12121@163.com 1232}
    // fl.Param():  1234
}

func ValidateErr(err error) any {
    errs, ok := err.(validator.ValidationErrors)
    if !ok {
        return err.Error()
    }
    var m = map[string]any{} // tag: 改造成返回map 把字段名翻译成对应label的中文名
    for _, e := range errs {
        msg := e.Translate(trans)

        tmpArr := strings.Split(msg, "---") // 对应标签注册时的分隔符
        m[tmpArr[0]] = tmpArr[1]
    }
    return m
}

func main() {

    r := gin.Default()
    gin.SetMode(gin.ReleaseMode)

    r.POST("post1", func(c *gin.Context) {
        type User struct {
            Name  string `json:"name" binding:"required" label:"用户名x"`
            Email string `json:"email" binding:"required,email"`
            Ip    string `json:"ip" binding:"fip=1234" label:"ip地址"` // tag: 自定义规则fip
        }
        var user User
        err := c.ShouldBindJSON(&user)
        if err != nil {
            msgBytes, _ := json.Marshal(ValidateErr(err))
            // 会存在参数验证失败 内置和自定义验证规则 ValidateErr获取失败的原因
            response.BadRequest(c, user, &response.ResponseOption{
                Msg: string(msgBytes), // tag: 验证的字符串
            })
            return
        }
        response.Success(c, user, nil)
    })

    r.Run(":8080")
}
```

# 日志
默认gin.Logger()中间件
使用官方包日志中间件 slog 

- 默认全局logger  slog.SetDefault
1. 内部其实是去读一个包级别的全局变量（slog 包内部维护的 defaultLogger）。slog.SetDefault(jsonLogger) 做的事情就是把这个全局变量指向你创建的 jsonLogger。 所以整个进程里，不管哪个包、哪个文件，只要用 slog.Info()、slog.Error() 这些包级别函数，拿到的都是同一个 logger 实例，不会创建新的
- traceId注入
``` go
// ContextHandler 从 context 中提取 TraceID 并添加到日志中 包一层slog.Handler
type ContextHandler struct {
    slog.Handler
}

func (h *ContextHandler) Handle(ctx context.Context, record slog.Record) error {
    if ctx != nil {
        if traceID, ok := ctx.Value(constant.TraceIDKey).(string); ok && traceID != "" {
            record.AddAttrs(slog.String(string(constant.TraceIDKey), traceID))
        }
    }
    return h.Handler.Handle(ctx, record)
}

func main() {
    // 初始化 slog  目前使用json 除了标准输出也可以文件
    baseHandler := slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
        Level: slog.LevelDebug, // 设置的日志等级 大于这个等级的才会打印
    })
    handler := &ContextHandler{Handler: baseHandler} // 包一层slog.Handler
    jsonLogger := slog.New(handler)
    // slog.SetDefault() 会把你创建的 logger 设置为全局默认logger。之后在项目任何地方直接调用 slog.Info()、slog.Error() 等包级别函数，都会走这个 JSON 格式的 logger，包括中间件和其他包里的调用。
    // 不调用 slog.SetDefault()，那这个 logger 就只是一个局部变量，只有显式传递 jsonLogger.Info(...) 才
    slog.SetDefault(jsonLogger)
    ....
}
```
